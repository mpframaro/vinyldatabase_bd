---///---
CRIACAO TABELAS
---///---





---///---COLECIONADORES---///---
create table Colecionadores
            (bi int primary key,
             nome varchar(50) not null,
             dataNascimento date not null,
             dataInicioColecao date);



---///---INTERPRETES---///---
create table Interpretes
            (codigoInterprete varchar(100) primary key,
             nome varchar(50) not null,
             origem varchar(50) not null,
             grammys int not null);



---///---PROJETOS---///---
create table Projetos
            (codigoProjeto varchar(100) primary key,
             nome varchar(100) not null,
             dataLancamento date not null,
             generos varchar(100) not null,
             tipo varchar(100) not null,
             formatos varchar(100) not null,
             reviewPitcfork int,
             codigoInterprete int references Interpretes(codigoInterprete));



---///---MUSICAS---///---
create table Musicas
            (codigoMusica varchar(100) primary key,
             nome varchar(100) not null,
             codigoProjeto int references Projetos(codigoProjeto));



---///---POSSUI---///---
create table Possui
            (bi int references Colecionadores(bi),
             codigoProjeto varchar(100) references Projetos(codigoProjeto),
             dataCompra date,
             primary key (bi, codigoProjeto));



---///---FEZ---///---
create table Fez
            (codigoMusica varchar(100) primary key references Musica(codigoMusica),
             codigoInterprete varchar(100) references Interprete(codigoInterprete));
